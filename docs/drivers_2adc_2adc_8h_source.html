514 
515  Adc *const adc_module = module_inst->hw;
516 
517  uint32_t int_flags = adc_module->INTFLAG.reg;
518 
519  uint32_t status_flags = 0;
520 
521  /* Check for ADC Result Ready */
522  if (int_flags & ADC_INTFLAG_RESRDY) {
523  status_flags |= ADC_STATUS_RESULT_READY;
524  }
525 
526  /* Check for ADC Window Match */
527  if (int_flags & ADC_INTFLAG_WINMON) {
528  status_flags |= ADC_STATUS_WINDOW;
529  }
530 
531  /* Check for ADC Overrun */

533  status_flags |= ADC_STATUS_OVERRUN;
