624 static inline void adc_set_gain(
625  struct adc_module *const module_inst,
626  const enum adc_gain_factor gain_factor)
627 {
628  /* Sanity check arguments */
629  Assert(module_inst);
630  Assert(module_inst->hw);
631 
632  Adc *const adc_module = module_inst->hw;
633 
634  while (adc_is_syncing(module_inst)) {
635  /* Wait for synchronization */
636  }
637 
638  /* Set new gain factor */
639  adc_module->INPUTCTRL.reg =
640  (adc_module->INPUTCTRL.reg & ~ADC_INPUTCTRL_GAIN_Msk) |
641  (gain_factor);

664 static inline enum status_code adc_set_pin_scan_mode(
665  struct adc_module *const module_inst,
666  uint8_t inputs_to_scan,
667  const uint8_t start_offset)
668 
669 {
670  /* Sanity check arguments */
671  Assert(module_inst);
672  Assert(module_inst->hw);
673 
674  Adc *const adc_module = module_inst->hw;
675 
676  if (inputs_to_scan > 0) {
677  /*
678  * Number of input sources included is the value written to INPUTSCAN
679  * plus 1.
680  */
681  inputs_to_scan--;
682  }
683 
684  if (inputs_to_scan > (ADC_INPUTCTRL_INPUTSCAN_Msk >> ADC_INPUTCTRL_INPUTSCAN_Pos) ||
685  start_offset > (ADC_INPUTCTRL_INPUTOFFSET_Msk >> ADC_INPUTCTRL_INPUTOFFSET_Pos)) {
686  /* Invalid number of input pins */
687  return STATUS_ERR_INVALID_ARG;
688  }
689 
690  while (adc_is_syncing(module_inst)) {
691  /* Wait for synchronization */
692  }
693 
694  /* Set pin scan mode */
695  adc_module->INPUTCTRL.reg =
696  (adc_module->INPUTCTRL.reg &
697  ~(ADC_INPUTCTRL_INPUTSCAN_Msk | ADC_INPUTCTRL_INPUTOFFSET_Msk)) |
698  (start_offset << ADC_INPUTCTRL_INPUTOFFSET_Pos) |
699  (inputs_to_scan << ADC_INPUTCTRL_INPUTSCAN_Pos);
700 
701  return STATUS_OK;
702 }
